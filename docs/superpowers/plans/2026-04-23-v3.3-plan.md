```python
_PRIVILEGED_INTENTS: frozenset[str] = frozenset({"members", "presences", "message_content"})
_logger = logging.getLogger("easycord")


def _collect_intents(obj: object) -> set[str]:
    if callable(getattr(obj, "get_intents", None)):
        return set(obj.get_intents())  # type: ignore[union-attr]
    result: set[str] = set()
    for cls in type(obj).__mro__:
        if "intents" in cls.__dict__:
            result |= set(cls.__dict__["intents"])
    return result
```

```python
def _ensure_intents(self) -> discord.Intents:
    if self._intents is None:
        self._intents = discord.Intents.default()
    return self._intents

def with_members(self) -> "Composer":
    """Enable the privileged Members intent (required for member join/leave events)."""
    self._ensure_intents().members = True
    return self

def with_messages(self) -> "Composer":
    """Enable the privileged Message Content intent (required to read message text)."""
    self._ensure_intents().message_content = True
    return self

def with_reactions(self) -> "Composer":
    """Enable the Reactions intent (required for reaction add/remove events)."""
    self._ensure_intents().reactions = True
    return self

def with_presence(self) -> "Composer":
    """Enable the privileged Presences intent (required for online status tracking)."""
    self._ensure_intents().presences = True
    return self

def with_voice(self) -> "Composer":
    """Enable the Voice States intent (required for voice channel join/leave events)."""
    self._ensure_intents().voice_states = True
    return self

def with_events(self) -> "Composer":
    """Enable the Guild Scheduled Events intent."""
    self._ensure_intents().guild_scheduled_events = True
    return self

def with_invites(self) -> "Composer":

    self._ensure_intents().invites = True
```

```python
def _resolve_plugin_intents(self) -> None:
    base = self._ensure_intents()
    for obj in (*self._plugins, *self._groups):
        for flag in _collect_intents(obj):
            if not getattr(base, flag, False):
                setattr(base, flag, True)
                if flag in _PRIVILEGED_INTENTS:
                    _logger.warning(
                        "Plugin %s requested privileged intent %r. "
                        "Ensure this is enabled in the Discord Developer Portal.",
                        type(obj).__name__,
                        flag,
                    )
```

```python
    # ── Invites ───────────────────────────────────────────────────

    async def create_invite(
        self,
        *,
        max_uses: int = 0,
        max_age: int = 0,
        temporary: bool = False,
        unique: bool = True,
        reason: str | None = None,
    ) -> discord.Invite:
        """Create an invite for the current channel.

        ``max_uses=0`` means unlimited; ``max_age=0`` means never expires.
        Raises ``RuntimeError`` if called outside a guild text channel.
        """
        if not isinstance(self.channel, discord.TextChannel):  # type: ignore[attr-defined]
            raise RuntimeError("create_invite requires a guild text channel")
        return await self.channel.create_invite(  # type: ignore[attr-defined]
            max_uses=max_uses,
            max_age=max_age,
            temporary=temporary,
            unique=unique,
            reason=reason,
        )

    async def get_invites(self) -> list[discord.Invite]:
        """Return all active invites for the current channel.

        Raises ``RuntimeError`` if called outside a guild text channel.
        """
        if not isinstance(self.channel, discord.TextChannel):  # type: ignore[attr-defined]
            raise RuntimeError("get_invites requires a guild text channel")
        return await self.channel.invites()  # type: ignore[attr-defined]

    async def revoke_invite(self, code: str) -> None:
        """Delete an invite by its code.

        Fetches the invite from Discord and deletes it. Raises
        ``discord.NotFound`` if the code is invalid.
        """
        invite = await self.interaction.client.fetch_invite(code)
        await invite.delete()
```

```python
"""Invite tracking plugin — tracks which invite new members used."""
from __future__ import annotations

import discord

from easycord import SlashGroup, on, slash


class InviteTrackerPlugin(SlashGroup, name="invite", description="Invite tracking and analytics"):
    """Track which invites members use when joining the server.

    Quick start::

        from easycord.plugins.invite_tracker import InviteTrackerPlugin

        bot = Composer().add_group(InviteTrackerPlugin()).build()

    Required intents: ``members`` (privileged), ``invites``.
    Required bot permission: ``MANAGE_GUILD`` (to list guild invites).

    Race condition note: if two members join at the exact same moment, attribution
    may be incorrect for one of them. This is a known limitation of the Discord API.
    """

    intents = {"members", "invites"}

    def __init__(self) -> None:
        super().__init__()
        self._snapshots: dict[int, dict[str, int]] = {}  # guild_id → {code: uses}

    # ── Internal helpers ──────────────────────────────────────────────────────

    async def _snapshot_guild(self, guild: discord.Guild) -> None:
        try:
            invites = await guild.invites()
        except discord.Forbidden:
            return
        self._snapshots[guild.id] = {inv.code: inv.uses or 0 for inv in invites}

    def _find_used(
        self, guild_id: int, current: list[discord.Invite]
    ) -> discord.Invite | None:
        old = self._snapshots.get(guild_id, {})
        for invite in current:
            if (invite.uses or 0) > old.get(invite.code, 0):
                return invite
        return None

    # ── Event handlers ────────────────────────────────────────────────────────

    @on("ready")
    async def _on_ready(self) -> None:
        for guild in self.bot.guilds:
            await self._snapshot_guild(guild)

    @on("guild_join")
    async def _on_guild_join(self, guild: discord.Guild) -> None:
        await self._snapshot_guild(guild)

    @on("member_join")
    async def _on_member_join(self, member: discord.Member) -> None:
        guild = member.guild
        try:
            current = await guild.invites()
        except discord.Forbidden:
            return
        used = self._find_used(guild.id, current)
        self._snapshots[guild.id] = {inv.code: inv.uses or 0 for inv in current}
        if used is None or used.inviter is None:
            return
        await self.bot.db.ensure_guild(guild.id)
        data: dict = await self.bot.db.get(guild.id, "invite_tracker") or {}
        inviter_key = str(used.inviter.id)
        data[inviter_key] = data.get(inviter_key, 0) + 1
        await self.bot.db.set(guild.id, "invite_tracker", data)
```

```python
    # ── Slash commands ────────────────────────────────────────────────────────

    @slash(description="Show how many members a user has invited", ephemeral=True)
    async def stats(self, ctx, member: discord.Member | None = None) -> None:
        data: dict = await self.bot.db.get(ctx.guild.id, "invite_tracker") or {}
        target = member or ctx.user
        uses = data.get(str(target.id), 0)
        await ctx.respond(
            f"**{target.display_name}** has invited **{uses}** member(s).",
            ephemeral=True,
        )

    @slash(description="Show the top inviters in this server")
    async def top(self, ctx, limit: int = 10) -> None:
        data: dict = await self.bot.db.get(ctx.guild.id, "invite_tracker") or {}
        if not data:
            await ctx.respond("No invite data recorded yet.", ephemeral=True)
            return
        ranked = sorted(data.items(), key=lambda x: x[1], reverse=True)[:limit]
        lines = [
            f"{i + 1}. <@{uid}> — **{count}** invite(s)"
            for i, (uid, count) in enumerate(ranked)
        ]
        await ctx.send_embed(
            title=f"Top {len(ranked)} Inviters",
            description="\n".join(lines),
        )
```
